// Kata 1 — makeNegative (ternarios anidados)
function makeNegative(numero) {
  return numero > 0 ? -numero
       : numero < 0 ?  numero
       : 0;
}

// Kata 2 — sumPositive (filter + reduce)
function sumPositive(arr) {
  const positivos = arr.filter(num => num > 0);
  return positivos.reduce((acc, num) => acc + num, 0);
}

  constructor(selector) {
    this.contenedor = document.querySelector(selector);
    this.track = this.contenedor.querySelector(".track");
    this.slides = this.contenedor.querySelectorAll(".slide");

    this.numero = this.contenedor.querySelector(".numero");
    
    this.indiceActual = 0;
    this.botonEscuchaActiva();
  }

  // El método que conecta estado con UI
  actualizarUI() {
    const desplazar = -(this.indiceActual * 100);
    this.track.style.transform = `translateX(${desplazar}%)`;

    document.querySelectorAll(".indicador")
      .forEach((punto, posicion) => {
        punto.classList.toggle(
          "active",
          posicion === this.indiceActual
        );
      });

    this.numero.innerText = this.indiceActual + 1;
  }

  siguiente() {
    this.indiceActual = (this.indiceActual + 1) % this.slides.length;
    this.actualizarUI();
  }

  anterior() {
    this.indiceActual = (this.indiceActual - 1 + this.slides.length) 
      % this.slides.length;
    this.actualizarUI();
  }

  botonEscuchaActiva() {
    this.botonPrev.addEventListener("click", this.anterior.bind(this));
    this.botonNext.addEventListener("click", this.siguiente.bind(this));
  }

Conceptos dominados
